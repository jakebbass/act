    <h1>Vie Capital Account Comparison Tool</h1>

    <form id="inputForm">

        <label for="paycheck">How much is your paycheck after taxes?</label>

        <input type="number" id="paycheck" placeholder="Enter amount in USD" required><br><br>

        

        <label for="frequency">How often are you paid?</label>

        <select id="frequency" required>

            <option value="Weekly">Every Week</option>

            <option value="Biweekly">Every 2 Weeks</option>

            <option value="Monthly">Once per Month</option>

            <option value="TwiceMonthly">Twice per Month</option>

        </select><br><br>

        

        <button type="button" onclick="calculateResults()">Calculate</button>

    </form>

    

    <h2>Comparison Chart</h2>

    <canvas id="comparisonChart" width="400" height="200"></canvas>

    

    <script>

        function calculateResults() {

            // Get user inputs

            const paycheck = parseFloat(document.getElementById('paycheck').value);

            const frequency = document.getElementById('frequency').value;


            // Annual pay calculation based on frequency

            let annualPay = 0;

            switch (frequency) {

                case "Weekly":

                    annualPay = paycheck * 52;

                    break;

                case "Biweekly":

                    annualPay = paycheck * 26;

                    break;

                case "Monthly":

                    annualPay = paycheck * 12;

                    break;

                case "TwiceMonthly":

                    annualPay = paycheck * 24;

                    break;

            }


            // Tool Calculator logic

            const checkingEarnings = annualPay;

            const checkingSpending = annualPay;

            const checkingLeftover = checkingEarnings - checkingSpending;


            const capitalEarnings = annualPay;

            const capitalSpending = annualPay * 0.9; // Example: 10% saving in spending

            const capitalLeftover = capitalEarnings - capitalSpending;


            // Render Chart

            renderChart([checkingLeftover, capitalLeftover]);

        }


        function renderChart(data) {

            const ctx = document.getElementById('comparisonChart').getContext('2d');

            new Chart(ctx, {

                type: 'bar',

                data: {

                    labels: ['Checking Account', 'Vie Capital Account'],

                    datasets: [{

                        label: 'Amount Leftover (USD)',

                        data: data,

                        backgroundColor: ['rgba(75, 192, 192, 0.2)', 'rgba(54, 162, 235, 0.2)'],

                        borderColor: ['rgba(75, 192, 192, 1)', 'rgba(54, 162, 235, 1)'],

                        borderWidth: 1

                    }]

                },

                options: {

                    scales: {

                        y: {

                            beginAtZero: true

                        }

                    }

                }

            });

        }

    </script>
